#include <iostream>
using namespace std;
int main()
{
    string input;
    cout<<"Enter a expression:";
    cin>>input;
    cout<<"operator found:\n";
    for(char op:input)
    {
        if(op=='+'||op=='-'||op=='*'||op=='/'||op=='%'||op=='=')
        {
            cout<<"operator:"<<op<<endl;
        }
    }
    return 0;
}
